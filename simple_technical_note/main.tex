%%% PRE-AMBLE %%%
\documentclass[11pt, a4paper, titlepage]{article}
\usepackage[a4paper, total={6in, 9in}]{geometry}    % sets margins

\usepackage{graphicx}   % need for figures
\usepackage{fancyhdr}   % used for custom header
\usepackage{lipsum}     % dummy text

%%% TITLE PAGE DETAILS %%%
\title{Technical Note}

\author{
    Jane Doe\\
    John Doe\\
}

% date and revision details
\date{\today}
\newcommand{\revision}{x.x}
\newcommand{\daterevised}{\today} % comment out this line to remove
\newcommand{\datesubmitted}{\today} % comment out this line to remove

% department
\newcommand{\department}{School of Electrical Engineering and Robotics}

% additional details about the technical note - comment out this line to remove
\newcommand{\technotedetails}{QCR Project Y - Note Z}


%%% DOCUMENT %%%
\begin{document}

    %%% TITLE PAGE %%%
    \input{title.tex}

    %%% CUSTOM HEADERS & FOOTERS %%%
    \pagestyle{fancy}
    
    \fancyhead{} % clear all header fields
    \fancyhead[L]{\textbf{\@title}\\\technotedetails}

    \fancyfoot{} % clear all footer fields
    \fancyfoot[C]{\thepage}

    %%% SECTIONS %%%
    \section*{Section X}
    \lipsum[2-4]

    \section*{Section Y}
    \lipsum[3-6]

    \section*{Section Z}
    \lipsum[7-8]

\end{document}
